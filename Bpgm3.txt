import matplotlib.pyplot as plt
from sklearn.datasets import load_iris
from sklearn.preprocessing import StandardScaler
from sklearn.decomposition import PCA
from sklearn.cluster import DBSCAN

# Step 1: Load data
X = load_iris().data

# Step 2: Standardize features
X = StandardScaler().fit_transform(X)

# Step 3: Reduce dimensions using PCA (2 components)
X_pca = PCA(n_components=2).fit_transform(X)

# Step 4: Apply DBSCAN clustering
labels = DBSCAN(eps=0.6, min_samples=5).fit_predict(X_pca)

# Step 5: Visualize clusters
plt.scatter(X_pca[:, 0], X_pca[:, 1], c=labels, cmap='plasma')
plt.title("DBSCAN Clustering after PCA")
plt.xlabel("PC1")
plt.ylabel("PC2")
plt.show()
